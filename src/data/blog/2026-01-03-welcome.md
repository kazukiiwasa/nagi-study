### Webix Widget デモ
ここにWebixのデータテーブルを埋め込んでみます。

<div id="webix_test_grid" style="width:100%; height:350px;"></div>

<link rel="stylesheet" href="https://cdn.webix.com/edge/webix.css" type="text/css">
<script src="https://cdn.webix.com/edge/webix.js" type="text/javascript"></script>

<script is:inline>
  const initWebixWidget = () => {
    const container = document.getElementById("webix_test_grid");
    // コンテナがない、または既に中身がある場合は二重表示防止のため中断
    if (!container || container.hasChildNodes()) return; 

    if (typeof webix !== "undefined") {
      // 省略していた「view」や「data」の設定をしっかり記述します
      webix.ui({
        container: "webix_test_grid",
        view: "datatable",
        columns: [
          { id: "id", header: "ID", width: 50 },
          { id: "task", header: "タスク", fillspace: true },
          { id: "status", header: "ステータス", width: 100 }
        ],
        data: [
          { id: 1, task: "Astro + Webix 連携", status: "Success" },
          { id: 2, task: "二重表示の防止", status: "Applied" },
          { id: 3, task: "View Transitions 対応", status: "Done" }
        ]
      });
    } else {
      setTimeout(initWebixWidget, 100);
    }
  };

  // Astroの画面遷移に対応
  document.addEventListener('astro:page-load', initWebixWidget);
  
  // 初回アクセス時用
  if (document.readyState === 'complete') {
    initWebixWidget();
  } else {
    window.addEventListener('load', initWebixWidget);
  }
</script>
